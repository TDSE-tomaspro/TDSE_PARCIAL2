import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

@RestController
public class ProxyController {

    // Se configuran por variables de entorno en EC2: EX: http://ec2-instance-1:8081
    private static final String SERVER_ACTIVE = System.getenv("SERVER_ACTIVE");
    private static final String SERVER_PASSIVE = System.getenv("SERVER_PASSIVE");

    @GetMapping("/compute")
    public String proxyRequest(@RequestParam(value = "val") String value) {
        try {
            // Intento 1: Servidor Activo
            return callService(SERVER_ACTIVE + "/math?val=" + value);
        } catch (Exception e) {
            try {
                // Intento 2: Fallback al Pasivo
                return callService(SERVER_PASSIVE + "/math?val=" + value);
            } catch (Exception ex) {
                return "{\"error\": \"Ambos servicios están fuera de línea\"}";
            }
        }
    }

    private String callService(String urlString) throws Exception {
        URL url = new URL(urlString);
        HttpURLConnection con = (HttpURLConnection) url.openConnection();
        con.setRequestMethod("GET");
        con.setConnectTimeout(2000); // Timeout corto para detectar caída rápido

        if (con.getResponseCode() == HttpURLConnection.HTTP_OK) {
            BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
            String inputLine;
            StringBuilder response = new StringBuilder();
            while ((inputLine = in.readLine()) != null) {
                response.append(inputLine);
            }
            in.close();
            return response.toString();
        }
        throw new RuntimeException("Service failure");
    }
}



package co.edu.eci.arsw.exam.math;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;
import java.util.ArrayList;
import java.util.List;

@RestController
public class MathController {

    @GetMapping("/math")
    public MathResult compute(@RequestParam(value = "val") int val) {
        // Ejemplo: Calcular secuencia de cuadrados o lo que pida el problema
        List<Integer> sequence = new ArrayList<>();
        for (int i = 0; i <= val; i++) {
            sequence.add(i * i);
        }
        return new MathResult("cuadrados", sequence);
    }
}
